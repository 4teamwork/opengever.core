% LISTINGS

% make itemize more compact (set itemsep and topsep globally to 0)
\makeatletter
\def\@listI{\leftmargin\leftmargini%
  \parsep 0\p@ \@plus0\p@ \@minus\p@
  \topsep 0\p@  \@minus\p@
  \itemsep 0\p@ \@plus0\p@ \@minus\p@}
\let\@listi\@listI%
\def\@listii {\leftmargin\leftmarginii
  \labelwidth\leftmarginii
  \advance\labelwidth-\labelsep
  \topsep    0\p@ \@plus0\p@ \@minus\p@
  \parsep    0\p@   \@plus\p@  \@minus\p@
  \itemsep   \parsep}
\makeatother
